  /**
  * Suma todos los valores del array pasado como argumento.
  *
  * @param[] $ar Array dde enteros que se quieren sumar.
  *
  *@return int el resultado de sumas todos los números del array de entrada.
  */

  function sumaArray($ar){
    $sum = 0;
    foreach ($ar as $n) {
      $sum+=$n;
    }
    return $sum;
  }
